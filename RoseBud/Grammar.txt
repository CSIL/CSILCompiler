program:
	| entry imports functions
	| entry functions
	| imports functions
	| functions
	;

entry:
	| "entry" IDENTIFIER ';'
	;

functions:
	| function
	| function functions
	;

function:
	| functionDefinition '{' statements '}'
	| functionDefinition '{' statements returnStatement '}'
	;

functionDefinition:
	| IDENTIFIER '(' ')'
	| IDENTIFIER '(' declarationList ')'
	;

declarationList:
	| "int" IDENTIFIER
	| "int" IDENTIFIER ',' declarationList
	;

statement:
	| IDENTIFIER = expression ';'
	;

returnStatement:
	| "return" expression ';'
	;

expression: term
	| term '+' term
	| term '-' term
	;

term: factor
	| factor '*' factor
	| factor '/' factor
	| factor '%' factor
	;

factor:
	| NUMBER
	| '(' expression ')'
	| functionCall
	;

functionCall:
	| IDENTIFIER '(' ')'
	| IDENTIFIER '(' argumentList ')'
	;

argumentList:
	| IDENTIFIER
	| expression
	;

imports:
	| import
	| import imports
	;

import:
	| "import" moduleName ';'
	;

moduleName:
	| IDENTIFIER
	| IDENTIFIER '.' moduleName
	;
